Centos7挂载物理分区

在host磁盘管理中添加分区

（WD 3TB为GPT分区方式，NTFS文件系统）

给vm添加物理磁盘，使用单个分区

（该分区为独占分区，一旦指定，vm和host不能同时访问）

- PhysicalDrive0 --> SSD 128G

- PhysicalDrive1 --> WD 1TB

- PhysicalDrive2 --> WD 3TB

- PhysicalDrive3 --> My Passport 1TB

vm重启后发现disk信息

[root@oracle1 ~]# lsblk
NAME            MAJ:MIN RM   SIZE RO TYPE MOUNTPOINT
sda               8:0    0    20G  0 disk
├─sda1            8:1    0   500M  0 part /boot
└─sda2            8:2    0  19.5G  0 part
  ├─centos-root 253:0    0  17.5G  0 lvm  /
  └─centos-swap 253:1    0     2G  0 lvm  [SWAP]
sdb               8:16   0   2.7T  0 disk
├─sdb1            8:17   0   128M  0 part
└─sdb2            8:18   0 488.3G  0 part
sr0              11:0    1  1024M  0 rom

centos 7下用ntfs-3g挂载

（http://www.ha97.com/2832.html）

[root@oracle1 ~]# yum install *ntfs*
[root@oracle1 ~]# rpm -a -q |grep ntfs
ntfs-3g-devel-2016.2.22-3.el7.x86_64
ntfs-3g-2016.2.22-3.el7.x86_64
ntfsprogs-2016.2.22-3.el7.x86_64
[root@oracle1 ~]# mkdir /mnt/F
[root@oracle1 ~]#  chown -R oracle:oinstall /mnt/F
[root@oracle1 ~]#  chown -R oracle:dba /mnt/F
[root@oracle1 ~]# cd /mnt
[root@oracle1 mnt]# ls
cdrom  F  hgfs
[root@oracle1 mnt]# mount -t ntfs-3g /dev/sdb2 /mnt/F  #挂载（临时，重启后失效）
[root@oracle1 mnt]# df -h
Filesystem               Size  Used Avail Use% Mounted on
/dev/mapper/centos-root   18G  9.8G  7.7G  56% /
devtmpfs                 903M     0  903M   0% /dev
tmpfs                    912M     0  912M   0% /dev/shm
tmpfs                    912M  8.6M  904M   1% /run
tmpfs                    912M     0  912M   0% /sys/fs/cgroup
/dev/sda1                497M  152M  345M  31% /boot
tmpfs                    183M     0  183M   0% /run/user/0
/dev/sdb2                489G  101M  489G   1% /mnt/F
[root@oracle1 mnt]# umount /dev/sdb2  #卸载
[root@oracle1 mnt]# vim /etc/fstab  #开机自动挂载
[root@oracle1 mnt]# cat /etc/fstab
# physical driver
/dev/sdb2 /mnt/F ntfs-3g defaults 0 0  #rw
